Linked list


Pre requisites:

  • Structures

  • Classes

  • Constructors

  • Pointers


There are different ways to implement, by using Structures and classes or just classes. Using a head pointer and tail pointer both or using head pointer only. Here we are gonna use Structures and classes both and only head pointer.

We will see how to make a Linked list and insert elements at rear, front and at a specific position and delete from front, rear and at a specific position and printing the size of the linked



An element is called a node in linked list, here we create a node using structure.

                struct node
                {
                    int data;
                    node *next;
                };

We define all the functions related to our linked list in a class. The code is pretty self explanatory with the comments.

    class ll
    {
    
    private:
        node *head;
        int nodes_present;
    
    public:
        ll()
        {
            head=NULL;
            nodes_present=0;
        }
        int size_of_ll()
        {
            return nodes_present;
        }
    
        void insert_rear(int n)
        {
            node *temp=new node;  //by new node we are making a new node whose address is going into the temp variable
            temp->data=n;
            temp->next=NULL;
    
            if(head==NULL)
                head=temp;
    
            else
            {
                node *ptr=head;
    
                while(ptr->next!=NULL)  //goes till the last node
                    ptr=ptr->next;
                ptr->next=temp;        //on next of last node, appends temp
            }
            nodes_present++;
        }
    
        void insert_front(int n)
        {
            node *temp=new node;
            temp->data=n;
            temp->next=NULL;
    
            if(head==NULL)
                head=temp;
            else
            {
                temp->next=head;
                head=temp;
            }
            nodes_present++;
        }
    
        void insert_at_pos(int n, int pos)
        {
    
            if(pos==0)
                insert_front(n);
            if(pos==(nodes_present))
                insert_rear(n);
            else
            {
                if(pos<1 || pos>nodes_present)
                    cout<<"invalid position\n";
                else
                {
                    node *temp=new node;
                    temp->data=n;
    
                    node *ptr=head;
                    while(pos--)
                    {
                        if(pos==0)
                        {
                            temp->next=ptr->next;
                            ptr->next=temp;
                        }
                    }
                }
            }
        }
    
    
        void del_rear()
        {
            if(head==NULL)
                cout<<"Linkedlist is empty\n";
            else
            {
                node *ptr=head;
                while(ptr->next->next !=NULL) //ptr goes till second last node
                    ptr=ptr->next;
                delete ptr->next;
                ptr->next=NULL;
                nodes_present--;
            }
    
        }
    
        void del_front()
        {
            if(head==NULL)
                cout<<"Linkedlist is empty\n";
            else
            {
                head=head->next;
                nodes_present--;
            }
        }
    
        void del_at_pos(int pos)
        {
            if(pos==0)
                del_front();
            else if(pos==(nodes_present-1))
                del_rear();
            else
            {
                node *ptr=head;
                while(pos--)
                {
                    if(pos==0)
                    {
                        ptr->next=ptr->next->next;
                    }
                }
            }
        }
    
        void display()
        {
            if(head==NULL)
                cout<<"Linkedlist is empty \n";
    
            else
            {
                node *ptr=head;
    
                while(ptr!=NULL)  //loop breaks after the last node
                {
                    cout<<ptr->data<<" ";
                    ptr=ptr->next;
                }
                cout<<"\n";
            }
        }
    };
