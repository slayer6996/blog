Queue


A Queue is First in Last Out (FIFO) datastructure. It can be implemented using arrays or linkedlist. Here we will see how to implement it using circular arrays.


A circular array can be thought of as a never ending array, basically which has its end and beginning connected.


A queue has following operations:

  • enqueue
  • dequeue
  • get front
  • get rear

All these operations are performed with O(1) time complexity.


Enqueue inserts an element into the queue from rear.

Dequeue deletes an element from the queue from front.

Get front returns the element which is at the front of the queue.

Get rear returns the element which is at the end of the queue.


To implement queue using circular array, while doing enqueue operation we modify the rear as (rear+1)%capacity and while doing dequeue operation we front one step ahead, for doing that, we use (front+1)%capacity .


The following code is C++ implementation of queue using circular array.

    //implementing queue with circular array

    struct Queue
    {
        int *ar;
        int Rear,Front;
        int size,capacity;
    
        Queue(int c)
        {
            capacity=c;
            ar=new int[capacity];
            Front=0;
            Rear=-1;            //it has the value -1 when queue is empty 
            size=0;
        }
    
        bool isempty()
        {
            return (size==0);
        }
    
        bool isfull()
        {
            return (size==capacity);
        }
    
        void enqueue(int x)
        {
            if(isfull())
                cout<<"overflow\n";
            else
            {
                Rear=(Rear+1)%capacity;
                ar[Rear]=x;
                size++;
            }
        }
    
        void dequeue()
        {
            if(isempty())
                cout<<"underflow\n";
            else
            {
                Front=(Front+1)%capacity;
                size--;
            }
        }

        void get_front()
        {
            if(isempty())
                cout<<"empty queue\n";
            else
            cout<<ar[front];
        }

        void get_rear()
        {
            if(isempty())
                cout<<"empty queue\n";
            else
            cout<<ar[rear];
        }
    
        void display()
        {
            if(isempty())
                cout<<"empty\n";
            else
            {
                if(Rear>=Front)
                {
                    for(int i=Front;i<=Rear;i++)
                        cout<<ar[i]<<" ";
                    cout<<"\n";
                }
                else
                {
                    for(int i=Front;i<size;i++)
                        cout<<ar[i]<<" ";
                    for(int i=0;i<=Rear;i++)
                        cout<<ar[i]<<" ";
                    cout<<"\n";
                }
            }
        }
    };
